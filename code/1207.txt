#include <stdio.h>
int main()
{
    int a, b, c, d, s;
    scanf("%d %d %d %d", &a, &b, &c, &d);
    s = a + b + c + d;
    if (s == 4)
        printf("À·");
    else if (s == 3)
        printf("°É");
    else if (s == 2)
        printf("°³");
    else if (s == 1)
        printf("µµ");
    else
        printf("¸ð");
    return 0;
}
