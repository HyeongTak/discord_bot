#include<stdio.h>

int main()
{  
    int a, b;
    scanf("%d %d", &a, &b);
     
    if(a % 2 == 1)
        printf("汝");
    else
        printf("礎熱");
     
    printf("+");
     
    if(b % 2 == 1) 
        printf("汝");
    else
        printf("礎熱");
     
    printf("=");
     
    if( (a+b) % 2 == 1)
         printf("汝");
    else
        printf("礎熱");
    return 0;
}
